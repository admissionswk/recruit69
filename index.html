/** * วางโค้ดนี้ใน Google Apps Script ของคุณ * โค้ดนี้อัปเดตให้สแกนข้อมูลเริ่มจากแถวที่ 4 และตรวจสอบ/บันทึกข้อมูลเริ่มที่คอลัมน์ D (ช่อง D4) */ function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let params; try { params = JSON.parse(e.postData.contents); } catch (err) { return createResponse("error", "รูปแบบข้อมูลไม่ถูกต้อง"); } const category = params.category; // ชื่อชีตควรตรงกับค่าที่ส่งมาจาก HTML: M1_Out, M1_In, M4_Gen, M1_Talent const sheet = ss.getSheetByName(category); if (!sheet) { return createResponse("error", "ไม่พบชื่อชีต: " + category + " ในระบบ"); } const lock = LockService.getScriptLock(); try { // รอคิว 10 วินาที เพื่อป้องกันคนกดสมัครพร้อมกันแล้วได้คิวเดียวกัน lock.waitLock(10000); const data = sheet.getDataRange().getValues(); let targetRow = -1; let applicantId = ""; let roomNumber = ""; // 1. ตรวจสอบหา "ช่องว่าง" เริ่มจากแถวที่ 4 (ใน Array คือ index 3) for (let i = 3; i < data.length; i++) { let idValue = data[i][2]; // คอลัมน์ C (index 2) = เลขประจำตัว let checkValue = data[i][3]; // คอลัมน์ D (index 3) = คำนำหน้า หรือ ชื่อ (เช็คว่าช่องนี้ว่างไหม) // เงื่อนไข: มีเลขประจำตัวในคอลัมน์ C, เป็นตัวเลข, และคอลัมน์ D ยังว่างอยู่ if (idValue && !isNaN(idValue) && String(idValue).trim() !== "" && !checkValue) { targetRow = i + 1; // แปลง Array Index เป็น Row Number ของ Google Sheet (เริ่มนับที่ 1) applicantId = idValue; // ดึงเลขห้องสอบจากคอลัมน์ F (index 5) อ้างอิงจากไฟล์ปกติ roomNumber = data[i][5] || ""; break; // เจอคิวแรกสุดที่ว่างแล้ว ให้หยุดค้นหา } } const timestamp = new Date(); if (targetRow !== -1) { // 2. อัปเดตข้อมูลนักเรียนคนใหม่ลงไปตามคอลัมน์ เริ่มที่คอลัมน์ D if (category === "M1_Talent") { // --- สำหรับ ม.1 ความสามารถพิเศษ (โครงสร้างคอลัมน์ต่างจากปกติ) --- sheet.getRange(targetRow, 4).setValue(params.title + params.fullname); // คอลัมน์ D: ชื่อ (รวมคำนำหน้า) sheet.getRange(targetRow, 8).setValue("'" + params.citizenId); // คอลัมน์ H: เลขประจำตัวประชาชน sheet.getRange(targetRow, 9).setValue(params.oldSchool); // คอลัมน์ I: โรงเรียนเดิม sheet.getRange(targetRow, 12).setValue(timestamp); // คอลัมน์ L: เวลาที่สมัคร (บันทึกกันลืม) } else { // --- สำหรับ ม.1 ในเขต, นอกเขต, และ ม.4 ทั่วไป --- sheet.getRange(targetRow, 4).setValue(params.title); // คอลัมน์ D: คำนำหน้า sheet.getRange(targetRow, 5).setValue(params.fullname); // คอลัมน์ E: ชื่อ sheet.getRange(targetRow, 9).setValue("'" + params.citizenId); // คอลัมน์ I: เลขประจำตัวประชาชน sheet.getRange(targetRow, 10).setValue(params.oldSchool); // คอลัมน์ J: โรงเรียนเดิม sheet.getRange(targetRow, 13).setValue(timestamp); // คอลัมน์ M: เวลาที่สมัคร } } else { // กรณีถ้าคุณเตรียมเลขประจำตัวไว้จนเต็มแล้วไม่มีช่องว่างเลย return createResponse("error", "ขออภัย ที่นั่งเต็ม หรือไม่ได้เตรียมเลขประจำตัวในระบบ กรุณาติดต่อเจ้าหน้าที่"); } // 3. เตรียมข้อมูลที่จะส่งกลับไปแสดงในบัตรยืนยันที่หน้า Canvas (HTML) const resultData = { applicantId: applicantId, fullName: params.title + params.fullname, room: roomNumber, categoryName: getCategoryName(category) }; return createResponse("success", "ยืนยันการสมัครสำเร็จ", resultData); } catch (err) { return createResponse("error", "เกิดข้อผิดพลาด: " + err.message); } finally { lock.releaseLock(); } } /** * ฟังก์ชันช่วยแปลงรหัสประเภทเป็นชื่อภาษาไทย */ function getCategoryName(cat) { const names = { "M1_Out": "ม.1 นอกเขตพื้นที่บริการ", "M1_In": "ม.1 ในเขตพื้นที่บริการ", "M4_Gen": "ม.4 ทั่วไป", "M1_Talent": "ม.1 ความสามารถพิเศษ" }; return names[cat] || cat; } /** * ฟังก์ชันสร้าง JSON Response ส่งกลับไปที่หน้าเว็บ */ function createResponse(status, message, data = null) { const output = { status: status, message: message, data: data }; return ContentService.createTextOutput(JSON.stringify(output)) .setMimeType(ContentService.MimeType.JSON); }
