/** * วางโค้ดนี้ใน Google Apps Script ของคุณ * จัดการบันทึกข้อมูล แยกตาม 4 ชีต ตรวจสอบคิวที่ว่างอยู่ และคำนวณเลขห้องสอบ */ function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let params; try { params = JSON.parse(e.postData.contents); } catch (err) { return createResponse("error", "รูปแบบข้อมูลไม่ถูกต้อง"); } const category = params.category; // ชื่อชีตควรตรงกับค่าที่ส่งมาจาก HTML: M1_Out, M1_In, M4_Gen, M1_Talent const sheet = ss.getSheetByName(category); if (!sheet) { return createResponse("error", "ไม่พบชื่อชีต: " + category + " ในระบบ"); } // กำหนดเลขประจำตัวเริ่มต้น (First ID - 1) สำหรับกรณีไม่มี Template เตรียมไว้ const startIdMap = { "M1_In": 10000, "M1_Out": 20000, "M4_Gen": 40000, "M1_Talent": 50000 }; const lock = LockService.getScriptLock(); try { // รอคิวบันทึกข้อมูลสูงสุด 10 วินาที เพื่อป้องกันคนกดสมัครพร้อมกันแล้วได้เลขซ้ำ lock.waitLock(10000); const data = sheet.getDataRange().getValues(); let targetRow = -1; let applicantId = ""; let roomNumber = ""; let nextCount = 0; // 1. ตรวจสอบหา "ช่องว่าง" จากรายการแรกๆ ก่อน (เริ่มจากแถวที่ 3 เพราะมี Header 2 แถว) // อ้างอิงตามคอลัมน์: B (index 1) = เลขประจำตัว, D (index 3) = ชื่อ for (let i = 2; i < data.length; i++) { let idValue = data[i][1]; // ข้อมูลในคอลัมน์ B (เลขประจำตัวผู้สมัคร) let nameValue = data[i][3]; // ข้อมูลในคอลัมน์ D (ชื่อ-สกุล) // ถ้าคอลัมน์ "เลขประจำตัว" มีค่าแล้ว แต่ช่อง "ชื่อ" ยังว่างอยู่ แสดงว่าเป็นคิวต่อไปที่ว่าง if (idValue !== "" && (nameValue === "" || nameValue == null)) { targetRow = i + 1; // Google Sheet อ้างอิงแถวเริ่มที่ 1 จึงต้อง +1 applicantId = idValue; // ดึงเลขห้องสอบจากคอลัมน์ E (index 4) ถ้าไม่ได้ใส่ไว้จะคำนวณอัตโนมัติ (ห้องละ 35) roomNumber = data[i][4] || Math.ceil((i - 1) / 35); nextCount = data[i][0] || (i - 1); // ลำดับที่ จากคอลัมน์ A (index 0) break; } } const timestamp = new Date(); if (targetRow !== -1) { // 2A. เจอช่องที่เตรียมไว้: อัปเดตข้อมูลนักเรียนคนใหม่ลงไปแทนที่ช่องว่าง sheet.getRange(targetRow, 3).setValue(params.title); // คอลัมน์ C: คำนำหน้า sheet.getRange(targetRow, 4).setValue(params.fullname); // คอลัมน์ D: ชื่อ-สกุล sheet.getRange(targetRow, 7).setValue("'" + params.citizenId); // คอลัมน์ G: เลขบัตรประชาชน (ใส่ ' กันแสดงผลเพี้ยน) sheet.getRange(targetRow, 8).setValue(params.oldSchool); // คอลัมน์ H: โรงเรียนเดิม sheet.getRange(targetRow, 9).setValue(timestamp); // คอลัมน์ I: เวลาที่สมัคร } else { // 2B. ไม่เจอช่องว่างเลย (คนเต็ม Template แล้ว): ทำการเพิ่มแถวใหม่ต่อท้ายให้เลยอัตโนมัติ const lastRow = sheet.getLastRow(); let currentCount = lastRow - 2; if (currentCount < 0) currentCount = 0; nextCount = currentCount + 1; applicantId = (startIdMap[category] || 0) + nextCount; roomNumber = Math.ceil(nextCount / 35); // คอลัมน์: A=ที่, B=เลข, C=คำนำหน้า, D=ชื่อ, E=ห้องสอบ, F=อาคาร, G=เลขบัตรฯ, H=โรงเรียน, I=เวลา sheet.appendRow([ nextCount, applicantId, params.title, params.fullname, roomNumber, "-", "'" + params.citizenId, params.oldSchool, timestamp ]); } // 3. เตรียมข้อมูลที่จะส่งกลับไปแสดงในบัตรยืนยันที่หน้า Canvas (HTML) const resultData = { applicantId: applicantId, fullName: params.title + params.fullname, room: roomNumber, categoryName: getCategoryName(category) }; return createResponse("success", "ยืนยันการสมัครสำเร็จ", resultData); } catch (err) { return createResponse("error", "เกิดข้อผิดพลาด: " + err.message); } finally { lock.releaseLock(); } } /** * ฟังก์ชันช่วยแปลงรหัสประเภทเป็นชื่อภาษาไทย */ function getCategoryName(cat) { const names = { "M1_Out": "ม.1 นอกเขตพื้นที่บริการ", "M1_In": "ม.1 ในเขตพื้นที่บริการ", "M4_Gen": "ม.4 ทั่วไป", "M1_Talent": "ม.1 ความสามารถพิเศษ" }; return names[cat] || cat; } /** * ฟังก์ชันส่ง Response กลับไปที่หน้าเว็บ */ function createResponse(status, message, data = null) { const output = { status: status, message: message, data: data }; return ContentService.createTextOutput(JSON.stringify(output)) .setMimeType(ContentService.MimeType.JSON); }
